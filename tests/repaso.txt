mi_programa patito

variable i, j, k : entero
variable arreglo : entero[12]
variable matriz : entero[12][9]

mi_funcion fact(j : entero) : entero {
    variable i : entero
    i = j + (k - j*2 + j)
    condicion (j == 1){
        regresar(j)
    }
    alternativa {
        regresar (j * fact(j-1))
    }
}

mi_funcion inicia(y : entero) : vacio {
    variable x : entero
    x = 1
    mientras(x < 11){
        arreglo[x] = y * x
    }
}

iniciar() {
    leer(k)
    j = k * 2
    i = 1
    mientras(i < 11){
        arreglo[i] = arreglo[i] * fact(arreglo[i]-k)
        matriz[i][arreglo[i+k]*k] = arreglo[i]
    }
}
